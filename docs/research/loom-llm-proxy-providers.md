llm\\ndata: {...}\\n\\n\"\n\n        loop {\n            match self.inner.poll_next_unpin(cx) {\n                Poll::Ready(Some(Ok(bytes))) => {\n                let stream_event: LlmStreamEvent =\n                                serde_json::from_str(&data)?;\n                            let event = convert_stream_event(stream_event);\n                            return Poll::Ready(Some(Ok(event)));\n                        }\n                    }\n                }\n                Poll::Ready(Some(Err(e))) => return Poll::Ready(Some(Err(e.into()))),\n                Poll::Ready(None) => return Poll::Ready(None),\n                Poll::Pending => return Poll::Pending,\n            }\n        }\n    }\n}\n\n// SSE LlmStreamEvent::ToolCallDelta { call_id, tool_name, arguments_fragment } => {\n            LlmEvent::ToolCallDelta { call_id, tool_name, arguments_fragment }\n        }\n
